 19 function getRequest($request_name, $default_value = null, $method = "all")
 20 {
 21     $magic_quotes = ini_get("magic_quotes_gpc") ? true : false;
 22     $method = strtolower($method);
 23
 24     switch (strtolower($method)) {
 25     default:
 26     case "all":
 27         if (isset($_POST[$request_name])) {
 28             return $magic_quotes ? stripslashes($_POST[$request_name]) : $_POST[$request_name];
 29         } else if (isset($_GET[$request_name])) {
 30             return $magic_quotes ? stripslashes($_GET[$request_name]) : $_GET[$request_name];
 31         } else {
 32             return $default_value;
 33         }
 34         break;
 35
 36     case "get":
 37         if (isset($_GET[$request_name])) {
 38             return $magic_quotes ? stripslashes($_GET[$request_name]) : $_GET[$request_name];
 39         } else {
 40             return $default_value;
 41         }
 42         break;
 43
 44     case "post":
 45         if (isset($_POST[$request_name])) {
 46             return $magic_quotes ? stripslashes($_POST[$request_name]) : $_POST[$request_name];
 47         } else {
 48             return $default_value;
 49         }
